	% 使用tabular环境生成表格
	% 有一个必选参数，指定表格的内容对齐格式
	% l 左对齐 c 居中 r 右对齐 用\\结尾
	%  可以使用\\hline产生横线
	% 可以使用两个\\hline和两个 || 产生双竖线 
	\begin{tabular}{|l|c|c|c|r|}
	\hline
		姓名 & 语文 & 数学 & 操作系统 & 数据结构 \\
	\hline
		yuluo & 100 & 100 & 100 & 100 \\
	\hline
		huakai & 100 & 100 & 100 & 100 \\
	\hline
	\end{tabular}

	\begin{tabular}{|l|c|c|c|p{1.5cm}|}

		姓名 & 语文 & 数学 & 操作系统 & 数据结构 \\
